def update
  @user = current_user
  respond_to do |format|
    if @user.update(user_params)
      @successmsg = {'msg'=>'success'}
      format.html { render action: 'edit' }
      format.json { render json: @successmsg }
    else
      format.html { render action: 'edit' }
      format.json { render json: @user.errors, status: :unprocessable_entity }
    end
  end
end
